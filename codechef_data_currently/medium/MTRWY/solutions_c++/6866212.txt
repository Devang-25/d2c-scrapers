#include <bits/stdc++.h>
#define gc getchar_unlocked
#define fo(i,n) for(i=0;i<n;i++)
#define Fo(i,k,n) for(i=k;i<n;i++)
#define lli long long int
#define pii pair<int,int>
#define vi vector<int>
#define pb push_back
#define mp make_pair
#define mod 1000000007

using namespace std;


int Find(int d[],int n){
  int t=n, k;
  while(d[t] != t) t=d[t];
  while(d[n] != n) k=d[n], d[n]=t, n=k;
  return n;
}
int Union(int p[],int a,int b)
{
	a = Find(p,a);
	b = Find(p,b);
	if(a==b) return 0;
	p[a] = b;
	return 1;
}

int t,n,m,q,nm,msz;
int type[1000000];
int sz[1000000];
int p[1000000];
int x1[1000000];
int ye1[1000000];
int x2[1000000];
int y2[1000000];
int br[1000][1000];
int bd[1000][1000];
int main()
{
	scanf("%d",&t);
	int i;
	lli res;
	int j,k;
	while(t--)
	{
		scanf("%d %d %d",&n,&m,&q);
		res=0;
		fo(i,n) fo(j,m) bd[i][j]=br[i][j]=0;
		nm = n*m;
		fo(i,q)
		{
			scanf("%d",&type[i]);
			if(type[i]==1)
			{
				scanf("%d %d",&x1[i],&ye1[i]);
				x1[i]--; ye1[i]--;
				br[x1[i]][ye1[i]]++;
				
			}
			if(type[i]==2)
			{

				scanf("%d %d",&x1[i],&ye1[i]);
				x1[i]--; ye1[i]--;
				bd[x1[i]][ye1[i]]++;
			}
			if(type[i]==3)
			{
				scanf("%d %d %d %d",&x1[i],&ye1[i],&x2[i],&y2[i]);
				
				x1[i]--; ye1[i]--;
				x2[i]--; y2[i]--;
				
			}
		}
		
		fo(i,nm){ p[i]=i;sz[i]=0;}
		fo(i,n-1) fo(j,m) if(!bd[i][j]) Union(p,i*m+j,(i+1)*m+j);
		fo(i,n) fo(j,m-1) if(!br[i][j]) Union(p,i*m+j,i*m+j+1);
		msz = 0;
		fo(i,nm)
		{
			k = Find(p,i);
			sz[k]++;
			msz=max(msz,sz[k]);
		}
		for(i=q-1;i>=0;i--)
		{
			if(type[i]==1)
			{
				if(--br[x1[i]][ye1[i]]==0)
				{
					j = Find(p,x1[i]*m+ye1[i]);
					k = Find(p,x1[i]*m+ye1[i]+1);
					if(Union(p,j,k))
					{
						sz[k]+=sz[j];
						msz=max(msz,sz[k]);
					}
				}
			}
			if(type[i]==2)
			{
				if(--bd[x1[i]][ye1[i]]==0)
				{
					j = Find(p,x1[i]*m+ye1[i]);
					k = Find(p,x1[i]*m+ye1[i]+m);
					if(Union(p,j,k))
					{
						sz[k]+=sz[j];
						msz=max(msz,sz[k]);
					}
				}
			}
			if(type[i]==3)
			{
			
				j = Find(p,x1[i]*m+ye1[i]);
				k = Find(p,x2[i]*m+y2[i]);
				if(j==k) res++;
			
				
			}
			if(type[i]==4)
			{
				res+=msz;
			
			}
		
		}
		printf("%lld
",res);
	
	}
	
	return 0;
}
