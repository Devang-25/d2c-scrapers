#define N 10000010

ll n,m;
ll freq[N];

ll mul(ll a,ll b)
{
    ll c = (ll)(a)*(ll)(b);
    c%=m;
    while(c < 0) c+=m;
    return c;
}

ll add(ll a,ll b) {
    ll c = (ll)(a) + (ll)(b);
    while(c >= m) c-=m;
    return c;
}

void calc(ll v) {
    freq[0]=1;
    for(int i=1;i<=v+2;i++) {
        freq[i] = mul(freq[i-1],i);
    }
}

ll get_x(ll val) {
    ll c = (ll)(val) * (ll)(val+1);
    c >>= 1;
    c%=m;
    while(c < 0) c+=m;
    c = (ll)(c) * (ll)(val);
    c%=m;
    while(c < 0) c+=m;

    return c;
}

int main(int argc, char const *argv[])
{
    sc(n); sc(m);

    ll a[n+3];

    ll max1 = -1;
    for(int i=0;i<n;i++) {
        scanf("%lld",&a[i]);
        a[i]%=m;
        if(a[i] < 0) a[i]+=m;
        if(max1 < a[i]) max1= a[i];
    }

    calc(max1);

    ll ans = 0;
    for(int i=0;i<n;i++) {
        ans = add(ans , add(freq[a[i]+1]-1 , get_x(a[i])));
    }   

    printf("%lld
",ans);
    return 0;
}
