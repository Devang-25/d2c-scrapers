ll n,m;
ll fact[10000002];

void setfact(ll m) {
    fact[0]=1;
    for(int i=1;i<=m;i++) {
        fact[i]=(fact[i-1]*i)% m;
    }
}

int main(){

for(i=0;i<n;i++){

    sum+=fact[a+1]-1+(((a*a)%m)*(a+1)/2)%m;

}
