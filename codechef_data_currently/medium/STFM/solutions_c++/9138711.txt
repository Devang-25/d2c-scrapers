#include<stdio.h>
#include<algorithm>
long long int f[10000006],p[100000];
using namespace std;
int main()
{
	long long int mx, n, m, i, temp, res = 0, fact, lim;
	scanf("%lld%lld", &n, &m);
	mx = -1;
	for (i = 0; i < n; i++)
	{
		scanf("%lld", &p[i]);
		temp = p[i];
		if(temp%2)
		res = ((((temp%m)*((((temp%m)*(((temp + 1)/2) % m)) % m) )) % m) + res) % m;
		else
			res = ((((temp%m)*(((((temp/2)%m)*((temp + 1) % m)) % m) )) % m) + res) % m;
		if (mx < temp)
			mx = temp;
	}
	lim = min(m, mx);
	fact = 1;
	if(m!=1)
		f[1] = 1;
	else
		f[1] = 0;
	for (i = 2; i < lim; i++)
	{
		fact = (fact*i)%m;
		f[i] = (f[i - 1] + (i*fact) % m) % m;
	}
	if (lim != m)
	{
		fact = (fact*i) % m;
		f[lim] = (f[lim - 1] + (lim*fact) % m) % m;
	}
	for (i = 0; i < n; i++)
	{
		res = (res + f[p[i]]) % m;
	}
	printf("%lld
", res);
	getchar();
	getchar();
	return 0;
}
