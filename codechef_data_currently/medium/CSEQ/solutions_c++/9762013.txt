#include<iostream>
using namespace std;

#define MOD 1000003

long long power(long long x, long long n){
    if(n==0)
        return 1;
    long long p = power(x, n/2);
    if(n%2==0)
        return (p*p)%MOD;
    else
        return ((x%MOD)*((p*p)%MOD))%MOD;
}

long long ncr(long long n, long long r)
{
    long long ans = 1;
    for(int i=0;i<=r-1;i++)
    {
        ans = (ans * (n-i))%MOD;
        ans = (ans * (power(i+1, MOD-2)%MOD))%MOD;
    }
    return ans;
}

long long calculate(long long n, long long r)
{
    long long ans = 1, ni, ri;
    while(n>0 && r>0)
    {
        ni = n%MOD;
        ri = r%MOD;
        if(ni>=ri)
            ans = (ans * ncr(ni, ri))%MOD;
        else
            ans = 0;
        n /= MOD;
        r /= MOD;
    }
    return (ans-1+MOD)%MOD;
}

int main()
{
    ios_base::sync_with_stdio(false);
    int t;
    long long n, l, r, ans;
    cin>>t;
    while(t--)
    {
        cin>>n>>l>>r;
        ans = calculate(n+r-l+1, n);
        cout<<ans<<endl;
    }
    return 0;
}
