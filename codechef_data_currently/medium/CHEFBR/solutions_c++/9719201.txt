#include<iostream>
using namespace std;

#define MOD 1000000007
typedef long long ll;

ll DP[100][100][2];

int main()
{
    ll n, ans;
    ll A[100];

    cin>>n;
    for(int i=0;i<n;i++)
        cin>>A[i];

    for(int step=1;step<n;step++)
    {
        for(int i=0;i<n-step;i++)
        {
            int j=i+step;
            ll left=0, right=0;
            for(int k=i;k<j;k++)
                right = (right + DP[k][j-1][0])%MOD;

            for(int k=i;k<=j;k++)
            {
                right = (right - DP[k][j-1][0])%MOD;

                int x = -1;
                if(A[i]<0 && A[k]==-A[i])
                    x = 0;
                if(A[k]<0 && A[k]==-A[j])
                    x = 1;

                if(x!=-1)
                    DP[i][j][x] = (DP[i][j][x] + (( (left+1) * (right+1) )%MOD))%MOD;

                left = (left + DP[i][k][1])%MOD;
            }
        }
    }

    //for(int i=0;i<n;i++)
    //{
        //for(int j=0;j<n;j++)
            //cout<<DP[i][j][0]<<" ";
        //cout<<endl;
    //}  
    //cout<<endl;
    //for(int i=0;i<n;i++)
    //{
        //for(int j=0;j<n;j++)
            //cout<<DP[i][j][1]<<" ";
        //cout<<endl;
    //}  

    ans = 1;
    for(int i=0;i<n;i++)
        ans = (ans + DP[0][i][1])%MOD;

    cout<<ans<<endl;
    return 0;
}
