#!/usr/local/bin/python2.7
#fo=open("input.txt","r+")
T=int(raw_input());
#T=int(raw_input());
for k in range(0,T):
  NK=raw_input();
  N=int(NK.split(" ")[0]);
  K=int(NK.split(" ")[1]);
  inp=raw_input().split(" ");
  MIN=pow(2,31);count=0;
  for i in range(0,N-1):
    for j in range(i+1,N):
      temp=abs(int(inp[i])+int(inp[j])-K);
      #print temp
      if temp<MIN:
        MIN=temp;
        count=1;
      else:
        if temp==MIN:
          count=count+1;
          
  print MIN,count
