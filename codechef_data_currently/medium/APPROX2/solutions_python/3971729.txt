T=int(raw_input());
#T=int(raw_input());
for k in range(0,T):
  NK=raw_input().split(" ")
  N=int(NK[0])
  K=int(NK[1])
  inp=raw_input().split(" ")
  MIN=abs(int(inp[0])+int(inp[1])-K)
  count1=0
  temp=0
  for i in range(0,N-1):
    for j in range(i+1,N):
      temp=abs(int(inp[i])+int(inp[j])-K)
      #print temp
      if(temp<MIN):
        MIN=temp
        count1=1
      else:
        if(temp==MIN):
          count1=count1+1
          
  print MIN,count1
