#include <iostream>
#include <cstdio>
#include<math.h>
using namespace std;
long long int t,n,k,x;
int main()
{

    scanf("%d",&t);
    while(t--){
        scanf("%lld %lld",&n,&k);
        x=(pow(k-1,n-1));
        if(k==1 && n!=1){
            printf("1
");
        }
        else
            printf("%lld
",(x*k)%1000000007);
    }
    return 0;
}
