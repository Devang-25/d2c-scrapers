#include <bits/stdc++.h>
using namespace std;

#define FI(i,a,b) for(i=a;i<b;++i)
#define F(i,n) FI(i,0,n)

#define lli long long

int main ()
{
	int t, n; scanf("%d", &t);
	string s;
	int type, b, g, i, j;
	lli sum, temp;
	while (t--) {
		scanf("%d", &type);
		cin >> s;
		n = s.length();
		sum = 100000000;
		F(j, 2) {
			F(i, n) {
				if (s[i] == 'B') s[i] = 'G';
				else s[i] = 'B';
			}
			
			b = g = 0;
			F(i, n) {
				if (s[i] == 'B') b++;
				else g++;
			}
			if (b != g && b != g+1) continue;
			
			if (type == 0) {
				temp = 0;
				for (i = 0; i < n; i += 2) {
					if (s[i] != 'B') temp++;		
				}
				sum = min(sum, temp);
			} else {
				temp = 0;
				int x = 0, y = 0;
				int a[100001], b[100001];
				for (i = 0; i < n; i += 2) if (s[i] != 'B') a[x++] = i;
				for (i = 1; i < n; i += 2) if (s[i] != 'G') b[y++] = i;
				F(i, x) temp += abs(a[i]-b[i]);
				sum = min(sum, temp);
			}
		}
		
		if (sum == 100000000) sum = -1;
		printf("%lld
", sum);
	}
	return 0;
}
