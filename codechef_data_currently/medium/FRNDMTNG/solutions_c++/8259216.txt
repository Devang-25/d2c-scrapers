#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    int t1,t2,p1,p2;
    long double a;
    scanf("%d",&t);
    while(t--)
    {
        cout<<fixed<<showpoint<<setprecision(10);
        scanf("%d%d%d%d",&p1,&p2,&t1,&t2);
        a = 1-((p1-t1)*(p2-t2))/(p1*p2);
        cout<<a;
        cout<<endl;
    }
    return 0;
}
