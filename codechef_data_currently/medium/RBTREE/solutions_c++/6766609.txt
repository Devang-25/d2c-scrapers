#include<iostream>//0 indicates black and 1 indicates red
using namespace std;
int arr[10000000];
int main()
{
	int q,i,count=0,l,r;
	char ch[3];
	cin>>q;
	arr[1]=0;
	for(i=1;i<=100000;i++)
	{
		if(arr[i]==0)
		{
			arr[2*i]=1;
			arr[(2*i)+1]=1;
		}
		if(arr[i]==1)
		{
			arr[2*i]=0;
			arr[(2*i)+1]=0;
		}
	}
	while(q--)
	{
	  cin>>ch;
	  if(ch[1]=='i')
	  {
	  	for(i=1;i<=100000;i++)
	  	if(arr[i]==0)
	  	arr[i]=1;
	  	else
	  	arr[i]=0;
	  }
	  if(ch[1]=='b')
	  {
	  	cin>>l>>r;
	  	  do
	  	{
	  	if(l==r)
	  	{
	  		if(arr[l]==0)
	  		count++;
	  		break;
	  	}
	  	else
	  	if(l>r)
	  	{
	  		if(arr[l]==0)
	  		count++;
	  		l=l/2;
	  	}
	  	else
	  	if(r>l)
	  	{
	  		if(arr[r]==0)
	  		count++;
	  		r=r/2;
	  	}
	  }while(1);
	  cout<<count<<"
";
	}
	if(ch[1]=='r')
	  {
	  	cin>>l>>r;
	  	do
	  	{
	  	if(l==r)
	  	{
	  		if(arr[l]==1)
	  		count++;
	  		break;
	  	}
	  	else
	  	if(l>r)
	  	{
	  		if(arr[l]==1)
	  		count++;
	  		l=l/2;
	  	}
	  	else
	  	if(r>l)
	  	{
	  		if(arr[r]==1)
	  		count++;
	  		r=r/2;
	  	}
	  }while(1);
	  cout<<count<<"
";
	}
	count=0;
}
	return 0;
}
