#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
      int n,k,m,i,a[100005]={0},b[100005]={0},c[100005]={0},left[100005]={0};
      cin>>n>>k>>m;
      for(i=0;i<n;i++)
      {
          cin>>a[i];
      }
      for(i=0;i<n;i++)
      {
          cin>>b[i];
      }
      for(i=0;i<k;i++)
      {
          cin>>c[i];
      }
      for(i=k;i<k+m;i++)
      {
          cin>>c[i];
      }
      for(i=0;i<n;i++)
      {
        left[i]=a[i]-b[i];
      }
      sort(c,c+m+k);
      sort(left,left+n);
      int j=m+k-1;
        for(int i=n-1;i>=0;i--)
        {
           while(j>=0)
            {
              if(c[j]<=left[i])
              {
                  left[i]=left[i]-c[j];
                  j--;
                  break;
              }
              j--;
            }
        }
        int sum=0;
        for(int y=0;y<n;y++)
        {
            sum+=left[y];
        }
        cout<<sum<<endl;
      }
      return 0;
}
