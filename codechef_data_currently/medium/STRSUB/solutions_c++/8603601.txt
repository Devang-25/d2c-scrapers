#include<iostream>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		int n,k=0,q,r;
		cin>>n>>k>>q;
		int l[n];
		string s;
		cin>>s;
		for(int i=0;i<n;i++){
			l[i]=1;
			r=1-(int)(s[i]-'0');
			for(int j=i+1;j<n;j++){
					r+=1-(int)(s[j]-'0');
					if(r>k||(j-i+1-r>k))
					break;
					l[i]++;
			}
		}
		while(q--){
			int ll,r,c=0,j;
			cin>>ll>>r;
			for(int i=ll-1;i<r;i++){
				if(l[i]<=r-i)
				c+=l[i];
				else
				c+=r-i;
		}
		cout<<c<<endl;
		}
}}
