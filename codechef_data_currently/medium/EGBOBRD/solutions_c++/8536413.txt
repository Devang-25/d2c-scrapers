#include<iostream>
#include<cmath>
using namespace std;
long int dp1[100002];
long int dp2[100002];
long int n, k, i, t, temp;
long int a[100002];
int main(void)
{
    cin>>t;
    while(t--)
    {
        cin>>n>>k;
        for(i = 0; i < n; i++)
            cin>>a[i];
        if(a[0]%k == 0)
        {
            dp1[0] = a[0]/k;
            //if(a[0] != 0)
            dp2[0] = 0;
            //else
              //  dp2[0] = k;
        }
        else
        {
            dp1[0] = 1 + (a[0]/k);
            dp2[0] = k - (a[0]%k);
        }
        for(i = 1; i < n; i++)
        {
            if(a[i]==0)
            {
                dp1[i] = dp1[i-1];
                dp2[i] = dp2[i-1];
                continue;
            }
            if((a[i]%k)==0)
            {
                dp1[i] = dp1[i-1]+(a[i]/k);
                dp2[i] = dp2[i-1];
                continue;
            }
            temp = a[i];
            if(dp2[i-1]!=0)
            {
                a[i] = a[i]-(dp2[i-1]-1);
            }
            if(a[i]<=0)
            {
                dp1[i] = dp1[i-1];
                dp2[i] = (dp2[i-1]-1)-temp;
                continue;
            }
            if((a[i]%k) == 0)
            {
                dp1[i] = dp1[i-1] + (a[i]/k);
                dp2[i] = 0;
            }
            else
            {
                dp1[i] = dp1[i-1] + 1 + (a[i]/k);
                dp2[i] = k - (a[i]%k);
            }

        }
        cout<<dp1[n-1]<<"
";
    }
}
