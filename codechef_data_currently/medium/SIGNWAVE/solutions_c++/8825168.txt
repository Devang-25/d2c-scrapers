#include<vector>

int main(){
    int t,i,j;
    cin>>t;
    vector<long long int> arr(51);
    arr[0] = 1;
    for(i=1;i<=50;i++){
        arr[i] = 2 * arr[i-1];
    }
    arr[0] = 0;
    //cout<<arr[50];
    while(t--){
        long long int s,c,k,sum = 0;
        //cout<<arr[4];
        cin>>s>>c>>k;

        if(s>=k)
            sum = sum + arr[s-k+1] + 1;
        if(k == 1){
            if(s>=(c+1))
                ;
            else{
                if(s > 0)
                    sum = sum + arr[c] - arr[s-1];
                else
                    sum = sum + arr[c];
            }
            cout<<sum<<endl;
            continue;
        }

        //cout<<sum<<endl;
        //j = k;
        int index = s - k + 1;
        if((index > 0) && (c>index) && (index<=50)){
            sum = sum + arr[index];
        }
       cout<<sum<<endl;

    }
}
