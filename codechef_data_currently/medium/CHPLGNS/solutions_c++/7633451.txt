#include<iostream>
#include<cstring>
#include<stdlib.h>
using namespace std;

int main()
 {
  int t,n,nov,k=0,i,nop[100000]={0};
  long long aov[100001][2],sum1,sum2,area[100001];
  cin>>t;
  while(t--)
   {
     cin>>n;
     memset(nop, 0, sizeof(nop));
     k=0;
     int n1=n;
     while(n1--)
      {
        cin>>nov;
        sum1=sum2=0;
        for(i=0;i<nov;i++)
         {
            cin>>aov[i][0]>>aov[i][1];
         }
        aov[i][0]=aov[0][0];
        aov[i][1]=aov[0][1];
        for(i=0;i<nov;i++)
         {
            sum1+= (aov[i][0]*aov[i+1][1]);
            sum2+= (aov[i][1]*aov[i+1][0]);   
         }
        
        area[k++]= (abs(sum1-sum2))/2;   
        for(i=0;i<k-1;i++)
         {
           if(area[i]>area[k-1])
             nop[i]++;
           else 
             nop[k-1]++;
         }      
      }   
     for(n1=0;n1<n;n1++)
      {
        cout<<nop[n1]<<" ";                  
      }  
     //cout<<endl;
   }
  //while(2);
  return 0;         
 }
