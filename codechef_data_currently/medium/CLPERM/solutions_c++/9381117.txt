#include<bits/stdc++.h>
using namespace std;
long long summ(int k,int j)
{
	long ans = j*j;
	ans-=j;
	ans-=k*k;
	ans-=k;
	ans/=2;
	return ans;
}
int main()
{
	std::ios::sync_with_stdio(false);

	int t; cin>>t;
	long n,k;
	while(t--)
	{
		cin>>n>>k;
		long long sum=n*(n+1);
		sum/=2;
		if(k==0){
			(sum%2==0)?cout<<"Chef
":cout<<"Mom
";
			continue;
		}
		else
		{
			int kk[k];
			for(int i=0;i<k;i++)cin>>kk[i];
			sort(kk,kk+k);
			sum+=summ(0,kk[0]);
			for(int i=1;i<k;i++)
			{
				if(kk[i]-kk[i-1]==1)continue;
				if(kk[i-1]<=sum)
					sum+=summ(kk[i-1],kk[i]);
				else
					break;
			}
			(sum%2==0)?cout<<"Mom
":cout<<"Chef
";
		}
		
	}
}
