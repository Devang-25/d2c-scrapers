#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
#define MOD 1000000007
ll calculateSum(ll a, ll b)
{
    a--;
    ll A = (a*(a+1)) >> 1LL;
    ll B = (b*(b+1)) >> 1LL;
    return (B - A);
}
int main()
{
	#ifndef ONLINE_JUDGE
        freopen("input.txt","r",stdin);
        freopen("output.txt","w",stdout);
    #endif
    int T; ll N,K;
    scanf("%d",&T);
    while(T--)
    {
        scanf("%lld%lld",&N,&K);
        if(K == 0)
        {
            ll ans = calculateSum(1,N) + 1;
            if(ans % 2 == 0) printf("Mom
");
            else printf("Chef
");
            continue;
        }
        vector<ll> missing(K);
        for(int i = 0; i < K; i++) scanf("%lld",&missing[i]);
        sort(missing.begin(),missing.end());
        ll S = calculateSum(1LL,max(missing[0] - 1, 0LL));
        //printf("initial S : %lld
",S);
        int ind = 0;
        ll ans = -1;
        while(ind < K)
        {
            while(ind + 1 < K && missing[ind+1] == missing[ind] + 1LL) ind++;
            ll nextAvailable = missing[ind] + 1;
            if(nextAvailable > N) { ans = S + 1; break;}
            if(nextAvailable > S+1)
            {
                ans = S+1;
                break;
            }
            else
            {
                if(ind + 1 < K)
                {
                    //printf("	Sum : %lld
",calculateSum(nextAvailable,missing[ind + 1] - 1));
                    S += calculateSum(nextAvailable,missing[ind + 1] - 1);
                    ind++;
                }
                else
                {
                    ans = calculateSum(1LL,N) + 1LL;
                    break;
                }
            }
        }
        //printf("ans : %lld
",ans);
        if(ans % 2 == 0) printf("Mom
");
        else printf("Chef
");
    }
    #ifndef ONLINE_JUDGE
    cerr << "Time elapsed: " << 1.0 * clock() / CLOCKS_PER_SEC << " s.
";
    #endif
    return 0;
}
