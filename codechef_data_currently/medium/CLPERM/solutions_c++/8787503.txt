#include<iostream>
#include<vector>
#include<climits>
#include<algorithm>
using namespace std;

int main(){
    int t;
    cin>>t;
    while(t--){
        int n,k,i,j;
        cin>>n>>k;
        vector<int> arr(101,0);
        for(i=0;i<k;i++){
            int temp;
            cin>>temp;
            arr[temp] = 1;
        }
        vector<int> trimArray;
        for(i=1;i<=n;i++){
            if(!arr[i])
                trimArray.push_back(i);
        }
        int len = trimArray.size();
        int sum = (100*101)/2;
        vector<int> ans(sum+1,0);
        ans[0] = 1;
        for(i=0;i<len;i++){
                int temp = trimArray[i];
            for(j=sum;j>=0;j--){
                if(ans[j] == 1 && ((j+temp)<=sum))
                    ans[j+temp] = 1;
            }
        }
        for(i=1;i<=sum;i++){
            if(!ans[i]){
                if(i%2){
                    cout<<"Chef"<<endl;

                    break;
                }
                else{
                    cout<<"Mom"<<endl;
                    break;
                }
            }
        }
    }
}
