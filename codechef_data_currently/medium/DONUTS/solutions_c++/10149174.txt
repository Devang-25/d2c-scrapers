#include <iostream>
#include <cstdio>
#include <cmath>
#include <algorithm>

using namespace std;
typedef long long ll;
ll V[1000000];

int main(){
	ios::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);
	ll tc,n,m;
	cin>>tc;
	while(tc--){
		cin>>n>>m;
		if(m==1){
			cout<<"0"<<'
';
			continue;
		}
		for(ll i=0;i<m;i++){
			cin>>V[i];
		}
		sort(V,V+m);
		ll i=0,j=m-1;
		ll ans=0;
		while(i<j){
			//cout<<i<<" "<<j<<endl;
			ans++;
			if(V[i]!=1){
				V[j-1]=V[j]+V[j-1]+1;
				i++;
			}else{
				V[j]=V[j]+V[i];
				i++;
				j--;
			}
		}
		cout<<ans<<'
';
	}
}
