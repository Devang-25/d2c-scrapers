#include <bits/stdc++.h>
using namespace std;

long long int blank_in_row[1000000];

int main() {
	// your code goes here
	long int i,h,l,N,H,T,just_opened[1000000];
	long long int max=0,temp=0;
	//memset(arrayname,0, sizeof(arrayname));
	scanf("%ld",&T);
	while(T--){
		memset(just_opened,0,sizeof(just_opened));
		scanf("%ld%ld",&N,&H);
		for(i=0;i<N;i++){
			scanf("%ld %ld",&l,&h);
			just_opened[l]++;
			just_opened[h+1]--;
		}
		temp=0;
		for(i=0;i<N;i++){
			temp=temp+just_opened[i];
			blank_in_row[i]=temp;
		}
		temp=0;
		for(i=0;i<N;i++){
			blank_in_row[i]+=temp;	
			temp=blank_in_row[i];
		}
		temp=0;
		max=-1;
		for(i=0;i<=N-H;i++){
			if(max<blank_in_row[i+H-1]-temp)max=blank_in_row[i+H-1]-temp;
			temp=blank_in_row[i];
		}
		printf("%lld
",N*H-max);
		
	}
	return 0;
}
