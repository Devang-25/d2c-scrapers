#include<iostream>
#include<stdlib.h>
#define high 1000000007
using namespace std;
long int a[1111];
long int count[1111];
long int n,m,s;
void funchoice(int s,int index)
{
    if(!(0<=s&&s<n))
        return;
    else if(index<m)
    {
     funchoice(s-a[index],index+1);
     funchoice(s+a[index],index+1);
    }
    else
    {
        count[s]++;
        if(count[s]>=high)
            count[s]-=high;
    }
}
int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
        cin>>n>>m>>s;
        for(int j=0;j<m;j++)
            cin>>a[j];
        for(int j=0;j<n;j++)
            count[j]=0;
        funchoice(s-1,0);
        for(int j=0;j<n;j++)
            cout<<count[j]<<" ";
        cout<<endl;
    }
    return 0;
}
