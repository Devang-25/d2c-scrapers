int input[100000];

long long ri() {
  char c = gc();
  while(c<'0' || c>'9') c = gc();
  long long  ret = 0;
  while(c>='0' && c<='9') {

    ret = 10 * ret + c - 48;
    c = gc();
  }
  return ret;
}

}

int get_value(long long x)
{

    long sum=0,index;

        return input[x];
        else{
            while(x>100000)
            {
                index=x%100000;
                sum+=input[index];
                x=x/100000;
            }
            sum+=input[x];
            return input[sum];
        }

}
