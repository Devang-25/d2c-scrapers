#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

bool inline check(string s,int l)
{
	bool f=true;
	if(l%2==0)
	{
		for(int i=0,j=l-1;i<l/2 && j>=l/2;i++,j--)
		{
			if(s[i]!=s[j])
			{
				f=false;
				break;
			}
		}
	}
	else
	{
		for(int i=0,j=l-1;i<l/2 && j>l/2;i++,j--)
		{
			if(s[i]!=s[j])
			{
				f=false;
				break;
			}
		}
	}
	return f;
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s,sc;
		cin>>s;
		sc=s;
		int l = s.length();
		int f=0;
		for(int i=0;i<l;i++)
		{
			sc.erase(i,1);
			if(check(sc,l-1))
			{
				f=1;
				break;
			}
			sc=s;
		}
		if(f==1)
		{
			cout<<"YES
";
		}
		else
		{
			cout<<"NO
";
		}
	}
	return 0;
}
