#include<iostream>
using namespace std;
int main(){
	int t,k,i,a[26];
	cin>>t;
	int count=0;
	string s;
	while(t--){
		count=0;
		cin>>s;
		for(i=0;i<26;i++){
			a[i]=0;
		}
		for(i=0;i<s.length();i++){
			k=(int)s[i];
			k=k-97;
			a[k]=++a[k];
		}
		for(i=0;i<26;i++){
			if(a[i]%2!=0){
				++count;
			}
		}
		if(count>2)
			cout<<"NO";
		else
		cout<<"YES";
		cout<<endl;
	}
	return 0;
}
