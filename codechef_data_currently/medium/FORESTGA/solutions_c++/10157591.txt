#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n,i;
    long long int r,l,ans,max1,g,w,q,h,m;
    pair<long long,long long> a[100005];
    cin>>n>>w>>q;
    max1=0;
    for(i=0;i<n;i++)
    {
        cin>>h>>m;
        a[i].first=h;
        a[i].second=m;
    }
    //cout<<max1;
    l=0;
    r=1e5+5;
    while(l<r)
    {
      long long int mid=(l+r)/2;
        ans=0;
        for(i=0;i<n;i++)
        {
            g=a[i].first+mid*a[i].second;
            if(g>=q)
            {
                ans=ans+g;
            }
        }
      //  cout<<ans<<"ans"<<"
";
        if(ans>=w)
        {
        r=mid;
        }
        else
            if(ans<w)
        {
            l=mid+1;
        }
    }
        cout<<l<<"
";
    return 0;
}
