T = int(raw_input())

for case in range(T):
	n = int(raw_input())
	vals = []
	for num in range(n):
		name, saidn = [x for x in raw_input().split(" ")]
		saidn = int(saidn)
		vals.append((name, saidn))
	vals = sorted(vals, key=lambda x: x[1])
	avals = [x[1] for x in vals]
	minnow = 2*(10**9)+10
	minname = ""
	for v in vals:
		if avals.count(v[1])==1 and v[1]<minnow:
			minnow = v[1]
			minname = v[0]
	print minname if minname!="" else "Nobody wins."
