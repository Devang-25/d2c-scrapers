#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll c[100001][2];
bool equals (ll a[], ll b[]){
if(a[0] == b[0] && a[1]==b[1])
    return true;
return false;
}
int main(){
	ll t,n, minx[3], maxx[3], miny[3], maxy[3];
	for(scanf("%lld",&t); t--; ){
		scanf("%lld",&n);
		minx [0] = 1e9 + 1;
		maxx [0] = -(1e9 + 1);
		miny [1] = 1e9 + 1;
		maxy [1] = -(1e9 + 1);
		for(ll i=0;i<n;i++){
		scanf("%lld%lld",&c[i][0],&c[i][1]);
			if(minx[0] >= c[i][0]){
                minx[0] = c[i][0];
                minx[1] = c[i][1];
                minx[2] = i;
			}

			if(miny[1] >= c[i][1]){
                miny[0] = c[i][0];
                miny[1] = c[i][1];
                miny[2] = i;
			}

			if(maxx[0] <= c[i][0]){
                maxx[0] = c[i][0];
                maxx[1] = c[i][1];
                maxx[2] = i;
			}

			if(maxy[1] <= c[i][1]){
                maxy[0] = c[i][0];
                maxy[1] = c[i][1];
                maxy[2] = i;
			}
		}

		if(equals(minx,maxy) ){ //left-upper
			//SE
			printf("1
%lld SE
",minx[2]+1);
		}
		else if(equals(minx,miny)){ //left-bottom
            //NE
            printf("1
%lld NE
", minx[2]+1);
		}
		else if(equals(maxx,maxy)){ //right-upper
            //SW
            printf("1
%lld SW
", maxx[2]+1);
		}
		else if(equals(maxx,miny)){ //right-bottom
            //NW
            printf("1
%lld NW
", maxx[2]+1);
		}
		else if(minx[1]>maxx[1]){
            printf("2
%lld SE
", minx[2]+1);
            printf("%lld NW
", maxx[2]+1);
		}
		else{
            printf("2
%lld NE
", minx[2]+1);
            printf("%lld SW
", maxx[2]+1);
		}
	}
	return 0;
}
