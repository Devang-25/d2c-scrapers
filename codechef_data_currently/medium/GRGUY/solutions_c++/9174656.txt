#include <iostream>
using namespace std;
#define min(x,y) (x<y) ? (x):(y)
string s1,s2;
int main(int argc, char const *argv[])
{
	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		cin>>s1>>s2;
		int len=s1.length();
		int i;
		int s=0;
		for(i=0;i<len;i++)
		{
			if(s1[i]=='#' && s2[i]=='#')
				break;
		}
		if(i==len)
		{
			cout<<"Yes
";
			for(i=0;i<len-1;i++)
			{
				if(s1[i]=='.' && s2[i]=='#' && s1[i+1]=='#' && s2[i+1]=='.')
					s++;
				if(s1[i]=='#' && s2[i]=='.' && s1[i+1]=='.' && s2[i+1]=='#')
					s++;
			}
			cout<<s<<"
";
		}
		else
			cout<<"No
";
	}
	return 0;
}
