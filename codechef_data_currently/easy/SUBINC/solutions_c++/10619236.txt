#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,t,i,c;
    long long int x;
    cin>>t;
    while(t--)
    {
        cin>>n;
        x=0,c=0;
        int ar[n];
        for(i=0;i<n;i++)
        {
           cin>>ar[i];
           if(ar[i]<ar[i-1]&&i>0)
           {
               x+=c*(c+1)/2;
               c=1;
           }
           else c++;
        }
        if(c>0)
        x+=c*(c+1)/2;
        cout<<x<<endl;
    }
    return 0;
}
