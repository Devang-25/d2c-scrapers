def solve(A, n):
    count = n
    for i in xrange(1, n):
        j = i
        while A[j] >= A[j-1]:
            count += 1
            j -= 1
    return count

if __name__ == '__main__':
    t = input()
    for _ in xrange(t):
        n = input()
        A = map(int, raw_input().split())
        print solve(A, n)
