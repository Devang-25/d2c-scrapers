#include<iostream>
#include<sstream>
#include<algorithm>
#include<cstring>

using namespace std;
int main(){
long long t,n;
cin>>t;
/*
The number of trailing zeroes increase by every (N+5) number
5!=1 zero
10!=2 zero
15!=3 zero
20!=4 zero
25!=5 zero
30!=6 zero
..
*/

while(t--){
	long long count=0;
	cin>>n;
	while(n!=0){
        n/=5;
        count+=n;
	}
    cout<<count<<endl;
}
//system("pause");
return 0;
}
