#include <bits/stdc++.h>
using namespace std;

int main()
{
#ifdef gsdt
    freopen("input.txt","r",stdin);
#endif // gsdt

    int n,k,a[100005],trace[100005];
    double dp[100005];
    const long long oo=1000000007;

    cin>>n>>k;
    for(int i=1; i<=n; i++) cin>>a[i];
    dp[1]=log(a[1]);
    trace[1]=0;
    for(int i=2; i<=k; i++) dp[i]=log(a[i]*a[1]), trace[i]=1;
    for(int i=k+1; i<=n; i++)
    {
        dp[i]=INT_MAX;
        for(int j=1; j<=k; j++)
            if(log(a[i])+dp[i-j]<dp[i])
            {
                dp[i]=log(a[i])+dp[i-j];
                trace[i]=i-j;
            }
    }

    long long ans=a[n];
    int i=n;
    while(trace[i])
    {
        i=trace[i];
        ans=1LL*(ans*a[i])%oo;
    }
    cout<<ans;
    return 0;
}
